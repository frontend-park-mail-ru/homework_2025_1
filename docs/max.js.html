 * Функция, определяющая максимальное число в массиве
 * @param {Array<Number>} numbers - массив чисел
 * 
 * @example
 * // returns 3
 * max([1, 2, 3]);
 * 
 * @returns {Number}
 */
const max = numbers => Math.max(...numbers);
